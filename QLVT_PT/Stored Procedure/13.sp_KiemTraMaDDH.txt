CREATE PROCEDURE sp_KiemTraMaDDH
    @MADDH NVARCHAR(8)
AS
BEGIN
   
   DECLARE	@return_value int
	set @return_value=0
	if exists (SELECT MasoDDH FROM dbo.DatHang WHERE MasoDDH=@X)
		set @return_value=1
	else if exists (SELECT MasoDDH FROM LINK1.QLVT.dbo.DatHang WHERE MasoDDH=@X)
		set @return_value=1

	SELECT	'Return Value' = @return_value

	
END;
